"""
Observation and Collapse Modeling Script

This script implements and demonstrates the concept of observation and collapse in The Nothingness Effect.
It combines the countable infinity and uncountable infinity concepts through the flowpoint function.

Key Components:
1. Imports necessary functions from other modules.
2. Defines the observation_and_collaps function.
3. Implements the combined_function that integrates countable and uncountable infinity.

Usage:
    Import this module in other scripts to use the observation_and_collaps function.
"""

import os
import sys

# Adjust path for dependencies
parent_dir = os.path.dirname(os.path.dirname(os.path.abspath(__file__)))
sys.path.insert(0, parent_dir)

from flowpoint.flowpoint import fp
from countable_infinity.countable_infinity import countable_infinity
from uncountable_infinity.uncountable_infinity import uncountable_infinity

def observation_and_collaps():
    """
    Implements the observation and collapse process in The Nothingness Effect.

    Returns:
        function: A function that takes three arguments (x, y, z) and returns a tuple of two values.
        
    The returned function (combined_function) integrates:
      - Countable Infinity (ℵ₀) via countable_infinity,
      - Uncountable Infinity (2ℵ₀) via uncountable_infinity,
    and then applies the flowpoint function (fp) to each, yielding:

         O(fp(ℵ₀, 2ℵ₀)) = (fp(ℵ₀), fp(2ℵ₀))
         
    This models the theoretical process of observation and collapse.
    """
    def combined_function(x, y, z):
        ci_gen = countable_infinity(x, y, z)
        ui_gen = uncountable_infinity(x, y, z)
        return fp(next(ci_gen)), fp(next(ui_gen))
    return combined_function

import os
import sys
import csv
import time
from tqdm import tqdm
import numpy as np
import matplotlib.pyplot as plt
import matplotlib.animation as animation
from mpl_toolkits.mplot3d import Axes3D

# Adjust path for dependencies
current_dir = os.path.dirname(os.path.abspath(__file__))
parent_dir = os.path.dirname(current_dir)
sys.path.insert(0, parent_dir)

from observation_and_collaps import observation_and_collaps

def test_observation_and_collaps(n_tests=1000):
    """
    Runs the observation and collapse function for random inputs.

    Args:
        n_tests (int): Number of random tests to perform.
    
    Returns:
        list: A list where each element is [x, y, z, ci_result, ui_result].
    """
    results = []
    print(f"Starting observation and collapse test with {n_tests} tests.")
    for _ in tqdm(range(n_tests), desc="Testing Observation and Collapse", unit="test"):
        x = np.random.uniform(-100, 100)
        y = np.random.uniform(-100, 100)
        z = np.random.uniform(-100, 100)
        try:
            ci_result, ui_result = observation_and_collaps()(x, y, z)
            results.append([x, y, z, ci_result, ui_result])
        except Exception as e:
            print(f"Error for inputs ({x}, {y}, {z}): {e}")
    print(f"All {n_tests} tests completed.")
    return results

def save_results(results, filename):
    """
    Saves test results to a CSV file.

    Args:
        results (list): List of test results.
        filename (str): Path of the output CSV.
    """
    try:
        print(f"Saving results to CSV: {filename}")
        start_time = time.time()
        with open(filename, 'w', newline='') as csvfile:
            writer = csv.writer(csvfile)
            writer.writerow(['X', 'Y', 'Z', 'Countable Infinity Result', 'Uncountable Infinity Result'])
            for row in tqdm(results, desc="Saving CSV", unit="row"):
                writer.writerow(row)
        elapsed_time = time.time() - start_time
        print(f"CSV saved to {filename} in {elapsed_time:.2f} seconds.")
    except Exception as e:
        print(f"Failed to save CSV: {e}")

def visualize_observation_and_collaps(results, filename_static, filename_animation):
    """
    Generates static and animated visualizations of the test results.

    Args:
        results (list): Test results.
        filename_static (str): PNG file path for static plot.
        filename_animation (str): GIF file path for animated visualization.
    """
    if not results:
        print("No results to visualize.")
        return

    x_values = np.array([r[0] for r in results])
    y_values = np.array([r[1] for r in results])
    z_values = np.array([r[2] for r in results])
    ci_results = np.array([r[3] for r in results])
    ui_results = np.array([r[4] for r in results])

    # Static Plot: 3D scatter for both sets of results
    try:
        print(f"Creating static visualization: {filename_static}")
        start_time = time.time()
        fig = plt.figure(figsize=(15, 12))
        ax1 = fig.add_subplot(221, projection='3d')
        ax1.scatter(x_values, y_values, z_values, c=ci_results, s=5, alpha=0.5)
        ax1.set_title('Countable Infinity Results')
        ax1.set_xlabel('X')
        ax1.set_ylabel('Y')
        ax1.set_zlabel('Z')

        ax2 = fig.add_subplot(222, projection='3d')
        ax2.scatter(x_values, y_values, z_values, c=ui_results, s=5, alpha=0.5)
        ax2.set_title('Uncountable Infinity Results')
        ax2.set_xlabel('X')
        ax2.set_ylabel('Y')
        ax2.set_zlabel('Z')

        # 2D Plots for demonstration
        ax3 = fig.add_subplot(223)
        ax3.plot(x_values, ci_results, 'go', label="Countable")
        ax3.plot(x_values, ui_results, 'ro', label="Uncountable")
        ax3.set_title('X vs Results')
        ax3.set_xlabel('X')
        ax3.set_ylabel('Result')
        ax3.legend()

        ax4 = fig.add_subplot(224)
        ax4.plot(y_values, ci_results, 'go', label="Countable")
        ax4.plot(y_values, ui_results, 'ro', label="Uncountable")
        ax4.set_title('Y vs Results')
        ax4.set_xlabel('Y')
        ax4.set_ylabel('Result')
        ax4.legend()

        plt.tight_layout()
        plt.savefig(filename_static)
        plt.close(fig)
        elapsed_time = time.time() - start_time
        print(f"Static visualization saved to {filename_static} in {elapsed_time:.2f} seconds.")
    except Exception as e:
        print(f"Static visualization error: {e}")

    # Animated Visualization
    try:
        print(f"Creating animated visualization: {filename_animation}")
        start_time = time.time()
        fig = plt.figure(figsize=(12, 12))
        ax = fig.add_subplot(111, projection='3d')

        def update(frame):
            ax.clear()
            ax.set_xlim(np.min(x_values), np.max(x_values))
            ax.set_ylim(np.min(y_values), np.max(y_values))
            ax.set_zlim(np.min(z_values), np.max(z_values))
            ax.set_xlabel('X')
            ax.set_ylabel('Y')
            ax.set_zlabel('Z')
            ax.set_title('Observation and Collapse Test Results')

            current_x = x_values[:frame+1]
            current_y = y_values[:frame+1]
            current_z = z_values[:frame+1]
            current_ci = ci_results[:frame+1]
            current_ui = ui_results[:frame+1]
            ax.scatter(current_x, current_y, current_z, c=current_ci, s=5, alpha=0.5, label='Countable')
            ax.scatter(current_x, current_y, current_z, c=current_ui, s=5, alpha=0.5, label='Uncountable')
            ax.legend()

        ani = animation.FuncAnimation(fig, update, frames=len(results), interval=50, repeat=False)
        plt.tight_layout()
        ani.save(filename_animation, writer='pillow', fps=20)
        elapsed_time = time.time() - start_time
        print(f"Animated visualization saved to {filename_animation} in {elapsed_time:.2f} seconds.")
        plt.close(fig)
    except Exception as e:
        print(f"Animated visualization error: {e}")
